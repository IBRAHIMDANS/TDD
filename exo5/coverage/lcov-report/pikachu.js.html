export const pikachu = function (number) {
  if (!number) {
    throw new Error("Please number can not undefined");
  }
  if (typeof number === "number") {
    let mod3 = number % 3 === 0;
    let mod5 = number % 5 === 0;
    if (mod3 && mod5) {
      return "PikaChu";
    } else if (mod5) {
      return "Chu";
    } else {
      return "Pika";
    }
    // return f ? (b ? "PikaChu" : "Pika") : b ? "Chu" : number;
  }
  throw new Error("Please number must be a number");
};
